def show_available_books():
    # Create table with headers
    table = PrettyTable()
    table.field_names = ["Book ID", "Title", "Author"]

    # Append row to table for each available book
    for book in library_books:
        if book["available"]:
            table.add_row([book["id"], book["title"], book["author"]])
    
    # Show table
    print(table)
